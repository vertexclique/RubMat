def subset(gen, zerosbst)
    main = case (zerosbst)
    when zerosbst == 'y'
            rslt = ui_pow(2, gen) - 1
            return rslt
    when zerosbst == 'n'
            rslt = ui_pow(2, gen)
            return rslt
    else
            puts "Enter 'y' or 'n' char"
    end
    return main
end
